<script>
  document.addEventListener("DOMContentLoaded", function () {
    function updateSliderPosition(toggleContainer) {
      let slider = toggleContainer.querySelector("#tripleToggleDot");
      let checkedRadio = toggleContainer.querySelector(
        'input[type="radio"]:checked'
      );
      let radios = Array.from(
      toggleContainer.querySelectorAll('input[type="radio"]')
      );
      let radioIndex = radios.indexOf(checkedRadio);
      let positions = [
      "left-[4px]",
      "left-[74px]",
      "left-2/4",
      "left-[212px]",
      ];

      if (radioIndex >= 0 && radioIndex < positions.length) {
        slider.className =
        "absolute w-1/4 h-3/4 bg-white shadow-md rounded-full transition-all duration-150 ease-in-out " +
        positions[radioIndex];
      } else {
        console.error("Invalid radio index:", radioIndex);
      }
    }

  document
    .querySelectorAll(".msc-label-toggle-container")
    .forEach(function (toggleContainer) {
      toggleContainer
    .querySelectorAll(".msc-triple-toggle-input-label input")
    .forEach(function (radio) {
      radio.addEventListener("change", function () {
        updateSliderPosition(toggleContainer);
      });
    });
      updateSliderPosition(toggleContainer);
    });
  });
</script>

<script>
  function setMaxHeight() {
    const tableDiv = document.getElementById("tableDiv");
    const modal = document.getElementById("modal");
    const modalOne = document.getElementById("modalOne");

    if (tableDiv && modal) {
    tableDiv.style.maxHeight = modal.offsetHeight - 200 + "px";
    }
  }

  window.onload = function () {
    setMaxHeight();
    const modalOne = document.getElementById("modalOne");
    if (modalOne && !modalOne.classList.contains("hidden")) {
      setMaxHeight();
    }
  };

  window.onresize = setMaxHeight;

  document.addEventListener("DOMContentLoaded", function () {
    const modalElement = document.getElementById("modal");
    if (modalElement) {
      modalElement.addEventListener("shown.bs.modal", setMaxHeight);
    }
  });
</script>
